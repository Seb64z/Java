mkdir = Crear carpeta
cd proyecto = Ingresar a carpeta proyecto
cd .. = salir de carpeta
touch = crear archivos
nano = bloc de notas
rm -R = eliminar archivo
cat = ver  contenido archivo
clear = limpia pantalla terminal
ctrl + l = lo mismo ||
history = Listar comandos escritos en terminal

Comandos git

git clone = clonar repositorio donde esté
git status = ve estado de estación
git log = ve trackeos o commit
git add . = agrega todos los archivos
git commit -m "Segundo viaje" = trackeo, nombre clave
git push origin main = subir al server
git pull origin main = actualiza
